<script>
const meals = [
  {
    id: 'earlyMorning',
    title: '🌅 Early Morning',
    fixed: true,
    options: [{ name: 'Matcha tea, oats, and half a banana', protein: 6 }]
  },
  {
    id: 'breakfast',
    title: '🍳 Breakfast',
    options: [
      { name: 'Avocado toast and Greek yogurt', protein: 12 },
      { name: 'Chickpea pancakes', protein: 10 }
    ]
  },
  {
    id: 'lunch',
    title: '🥗 Lunch',
    options: [
      { name: 'Handmade paneer burger and side salad', protein: 20 },
      { name: 'Tofu wrap and side salad', protein: 18 }
    ]
  },
  {
    id: 'eveningSnack',
    title: '🍵 Evening Snack',
    fixed: true,
    options: [{ name: 'Matcha tea and roasted chickpeas', protein: 5 }]
  },
  {
    id: 'dinner',
    title: '🍛 Dinner',
    options: [
      { name: 'Chapatis, subzi, and dal', protein: 15 },
      { name: 'Roasted vegetable and paneer salad', protein: 12 }
    ]
  },
  {
    id: 'night',
    title: '🌙 Night',
    fixed: true,
    options: [{ name: 'Herbal tea', protein: 0 }]
  }
];

const todayKey = 'dietTracker_' + new Date().toISOString().slice(0, 10);
let state = JSON.parse(localStorage.getItem(todayKey) || '{}');

document.getElementById('date').textContent = new Date().toLocaleDateString('en-US', {
  weekday: 'long', year: 'numeric', month: 'long', day: 'numeric'
});

function render() {
  const container = document.getElementById('meals');
  container.innerHTML = '';
  meals.forEach(meal => {
    if (!state[meal.id]) state[meal.id] = { choice: 0, done: false };
    const div = document.createElement('div');
    div.className = 'meal';
    let html = `<div class="meal-title">${meal.title}</div>`;
    if (!meal.fixed) {
      html += `<select onchange="updateChoice('${meal.id}', this.value)">`;
      meal.options.forEach((o, i) => {
        html += `<option value="${i}" ${state[meal.id].choice == i ? 'selected' : ''}>${o.name} (${o.protein}g protein)</option>`;
      });
      html += `</select>`;
    }
    const opt = meal.options[state[meal.id].choice];
    html += `<div class="check-row ${state[meal.id].done ? 'done' : ''}">
      <input type="checkbox" id="chk_${meal.id}" ${state[meal.id].done ? 'checked' : ''} onchange="toggleDone('${meal.id}', this.checked)">
      <label for="chk_${meal.id}">${meal.fixed ? opt.name : 'Mark as eaten'} — ${opt.protein}g protein</label>
    </div>`;
    div.innerHTML = html;
    container.appendChild(div);
  });
  updateStats();
  save();
}

function updateChoice(id, val) {
  state[id].choice = parseInt(val);
  render();
}

function toggleDone(id, checked) {
  state[id].done = checked;
  render();
}

function updateStats() {
  let protein = 0, done = 0;
  meals.forEach(m => {
    if (state[m.id].done) {
      done++;
      protein += m.options[state[m.id].choice].protein;
    }
  });
  document.getElementById('protein').textContent = protein;
  document.getElementById('completed').textContent = done;
  document.getElementById('total').textContent = meals.length;
  document.getElementById('progressBar').style.width = (done / meals.length * 100) + '%';
}

function save() {
  localStorage.setItem(todayKey, JSON.stringify(state));
}

function resetDay() {
  if (confirm("Reset today's progress?")) {
    state = {};
    localStorage.removeItem(todayKey);
    render();
  }
}

render();
